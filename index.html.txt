    <script>
    document.addEventListener('DOMContentLoaded', () => {

        // --- 1. Entry Screen Logic ---
        const landingPage = document.getElementById('landing');
        const enterButton = document.getElementById('enter-button');
        const mainContent = document.getElementById('main-content');
        const body = document.body;

        enterButton.addEventListener('click', () => {
            // Fade out the landing page
            landingPage.classList.add('hidden');
            
            // Allow scrolling on the body
            body.style.overflowY = 'auto';
            
            // Fade in the main content after the landing page is gone
            mainContent.classList.add('visible');

            // Trigger the kinetic text animation
            kineticTextAnimation();
        });

        // --- 2. Kinetic Typography for Hero Title ---
        function kineticTextAnimation() {
            const title = document.getElementById('kinetic-title');
            if (!title) return;
            const text = title.textContent;
            title.innerHTML = ''; // Clear original text
            
            text.split('').forEach((letter, index) => {
                const span = document.createElement('span');
                span.textContent = letter === ' ' ? '\u00A0' : letter; // Handle spaces
                span.style.animationDelay = `${index * 0.05}s`;
                title.appendChild(span);
            });
        }
        
        // --- 3. Scroll-triggered Animations ---
        const animatedElements = document.querySelectorAll('.animate-on-scroll');

        const observer = new IntersectionObserver((entries) => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    entry.target.classList.add('is-visible');
                    // Optional: stop observing once it's visible so it doesn't re-animate
                    observer.unobserve(entry.target);
                }
            });
        }, {
            threshold: 0.1 // Trigger when 10% of the element is visible
        });

        animatedElements.forEach(element => {
            observer.observe(element);
        });

    });
    </script>
